### 4.1 `src/io/registry.py` —— 按名字选 loader

```python
from __future__ import annotations

from pathlib import Path
from typing import Callable

from src.io.load_bci2a_mat import load_bci2a_mat
from src.io.load_gdf import load_gdf
from src.io.load_openbci_csv import load_openbci_csv
from src.types import ContinuousEEG

# loader(path) -> ContinuousEEG | list[ContinuousEEG]
LoaderFn = Callable[[Path], ContinuousEEG | list[ContinuousEEG]]

REGISTRY: dict[str, LoaderFn] = {
    "load_bci2a_mat": load_bci2a_mat,
    "load_gdf": load_gdf,
    "load_openbci_csv": load_openbci_csv,
}


def get_loader(name: str) -> LoaderFn:
    if name not in REGISTRY:
        raise KeyError(f"未知 loader: {name}；可选: {list(REGISTRY)}")
    return REGISTRY[name]


def as_run_list(obj: ContinuousEEG | list[ContinuousEEG]) -> list[ContinuousEEG]:
    return obj if isinstance(obj, list) else [obj]
```

### 4.2 `src/config_load.py` —— 读 yaml

```python
from __future__ import annotations

from pathlib import Path
import yaml


def load_config(path: str | Path) -> dict:
    path = Path(path)
    with path.open("r", encoding="utf-8") as f:
        cfg = yaml.safe_load(f)
    # 轻量校验
    for key in ("dataset", "loader", "label_map", "target_chans", "out_dir"):
        if key not in cfg:
            raise KeyError(f"配置缺少字段: {key}")
    # yaml 里数字键可能是 int，统一成 int
    cfg["label_map"] = {int(k): tuple(v) for k, v in cfg["label_map"].items()}
    cfg["drop_codes"] = [int(c) for c in cfg.get("drop_codes", [])]
    return cfg
```

依赖：已有 `pyyaml`（`requirements.txt`）。

### 4.3 通用标签过滤（替代写死的 THREE_MAP）

可新建 `src/steps/harmonize_from_config.py`，或给现有函数加参数：

```python
import numpy as np


def filter_events_by_map(
    events: np.ndarray,
    artifacts: np.ndarray | None,
    label_map: dict[int, tuple[int, int]],
    drop_codes: list[int] | None = None,
) -> np.ndarray:
    """
    返回 (n_keep, 4): [cue, label_task, label_three, trial_index]
    label_map: 原始码 → (task, three)
    drop_codes: 额外丢弃（也可只靠「不在 label_map 里」丢弃）
    """
    drop = set(drop_codes or [])
    kept = []
    for i, (samp, y) in enumerate(events):
        y = int(y)
        if artifacts is not None and int(artifacts[i]) == 1:
            continue
        if y in drop or y not in label_map:
            continue
        task, three = label_map[y]
        kept.append([int(samp), int(task), int(three), i])
    if not kept:
        return np.zeros((0, 4), dtype=int)
    return np.asarray(kept, dtype=int)
```

BCI 2a 配置里的 `label_map` / `drop_codes` 就对应现在的 `{1→(1,1), 2→(1,2)}` 和丢弃 3、4。

### 4.4 `scripts/run_preprocess.py` —— 命令行入口示意

```python
"""
用法（在 code 目录，设好 PYTHONPATH=preprocess_lab）:

  python -m scripts.run_preprocess --config configs/bci2a.yaml

（落地后路径以实际包名为准）
"""
from __future__ import annotations

import argparse
from pathlib import Path
import glob

import numpy as np

from src.config_load import load_config
from src.io.registry import get_loader, as_run_list
from src.pipeline_unified import preprocess_run_cfg, save_outputs  # 名称示意


def collect_files(cfg: dict) -> list[Path]:
    if cfg.get("data_files"):
        return [Path(p) for p in cfg["data_files"]]
    pattern = cfg["data_glob"]
    files = sorted(Path(p) for p in glob.glob(pattern))
    if not files:
        raise FileNotFoundError(f"没有匹配到数据: {pattern}")
    return files


def main() -> None:
    ap = argparse.ArgumentParser()
    ap.add_argument("--config", required=True, help="如 configs/bci2a.yaml")
    args = ap.parse_args()

    cfg = load_config(args.config)
    loader = get_loader(cfg["loader"])
    out_dir = Path(cfg["out_dir"])
    out_dir.mkdir(parents=True, exist_ok=True)

    Xs, yts, y3s, subs = [], [], [], []
    for fpath in collect_files(cfg):
        runs = as_run_list(loader(fpath))
        subject = fpath.stem[:3]  # A01；其它数据集可在配置里写 subject_from: stem
        for eeg in runs:
            X, yt, y3 = preprocess_run_cfg(eeg, cfg)
            if len(yt) == 0:
                continue
            Xs.append(X)
            yts.append(yt)
            y3s.append(y3)
            subs.append(np.array([subject] * len(yt)))

    X = np.concatenate(Xs, axis=0)
    y_task = np.concatenate(yts, axis=0)
    y_three = np.concatenate(y3s, axis=0)
    subjects = np.concatenate(subs, axis=0)

    save_outputs(X, y_task, y_three, subjects, cfg)
    print("done", cfg["dataset"], X.shape)


if __name__ == "__main__":
    main()
```

### 4.6 整库自动处理：一个数据集里的「所有文件」（思路 + 完整示例）

#### 思路

```text
configs/bci2a.yaml
  data_glob: ".../A0*T.mat"
        │
        ▼
collect_files(cfg)  →  [A01T.mat, A02T.mat, ..., A09T.mat]
        │
        ▼
对每个文件:
    loader(path) → 多个 ContinuousEEG（多 run）
    对每个 run: preprocess_run_cfg(eeg, cfg)
    记下 subject（如文件名 A01）
        │
        ▼
concatenate 所有人的 X / y_task / y_three / subjects
        │
        ▼
save_full → 全量 npy
split_all_trials(8:2) → train/val npy
```

对比现状：

| | 现在的 `pipeline.py` | 本框架示例 |
|--|---------------------|------------|
| 单 `.mat` 内多 run | ✅ 已支持 | ✅ |
| 文件夹内全部被试文件 | ❌ 写死 A01 | ✅ `data_glob` 自动扫 |

#### `collect_files` + 整库主流程（完整示例代码）

```python
from __future__ import annotations

import argparse
import glob
from pathlib import Path

import numpy as np

from src.config_load import load_config
from src.io.registry import get_loader, as_run_list
from src.steps.split_subjects import split_all_trials
# preprocess_run_cfg / select 等见上文示意


def collect_files(cfg: dict) -> list[Path]:
    """根据配置收集一个数据集下的全部输入文件。"""
    if cfg.get("data_files"):
        files = [Path(p) for p in cfg["data_files"]]
    else:
        pattern = cfg["data_glob"]
        files = sorted(Path(p) for p in glob.glob(pattern))
    if not files:
        raise FileNotFoundError(
            f"没有匹配到数据文件。请检查 data_glob / data_files。cfg={cfg.get('dataset')}"
        )
    print(f"[{cfg['dataset']}] 共 {len(files)} 个文件:")
    for f in files:
        print(" ", f)
    return files


def subject_id_from_path(path: Path, cfg: dict) -> str:
    """默认用文件名前 3 位（A01）；其它数据集可在 yaml 里扩展规则。"""
    mode = cfg.get("subject_from", "stem3")
    if mode == "stem3":
        return path.stem[:3]
    if mode == "stem":
        return path.stem
    raise ValueError(f"未知 subject_from: {mode}")


def process_whole_dataset(cfg: dict) -> tuple[np.ndarray, np.ndarray, np.ndarray, np.ndarray]:
    """
    自动处理配置所指向的「整个数据集」全部文件。
    返回合并后的 X, y_task, y_three, subjects。
    """
    loader = get_loader(cfg["loader"])
    Xs, yts, y3s, subs = [], [], [], []

    for fpath in collect_files(cfg):
        sid = subject_id_from_path(fpath, cfg)
        runs = as_run_list(loader(fpath))
        print(f"  → {fpath.name}: subject={sid}, runs={len(runs)}")

        for eeg in runs:
            # 若 loader 已填 eeg.subject，也可用 eeg.subject
            X, yt, y3 = preprocess_run_cfg(eeg, cfg)
            if len(yt) == 0:
                continue
            Xs.append(X)
            yts.append(yt)
            y3s.append(y3)
            subs.append(np.array([sid] * len(yt)))

    if not Xs:
        raise RuntimeError("全部文件处理后没有任何有效试次")

    X = np.concatenate(Xs, axis=0)
    y_task = np.concatenate(yts, axis=0)
    y_three = np.concatenate(y3s, axis=0)
    subjects = np.concatenate(subs, axis=0)
    print(
        "合并完成:",
        "X", X.shape,
        "y_task", np.bincount(y_task, minlength=2),
        "y_three", np.bincount(y_three, minlength=3),
        "subjects", sorted(set(subjects.tolist())),
    )
    return X, y_task, y_three, subjects


def save_outputs(
    X: np.ndarray,
    y_task: np.ndarray,
    y_three: np.ndarray,
    subjects: np.ndarray,
    cfg: dict,
) -> Path:
    """按配置保存全量 + train/val。"""
    out_dir = Path(cfg["out_dir"])
    out_dir.mkdir(parents=True, exist_ok=True)
    tag = cfg["dataset"]

    if cfg.get("save_full", True):
        np.save(out_dir / f"{tag}_X.npy", X)
        np.save(out_dir / f"{tag}_y_task.npy", y_task)
        np.save(out_dir / f"{tag}_y_three.npy", y_three)
        np.save(out_dir / f"{tag}_subjects.npy", subjects)
        print("saved full →", out_dir)

    if cfg.get("save_split", True):
        parts = split_all_trials(
            X,
            y_task,
            y_three,
            val_ratio=float(cfg.get("val_ratio", 0.2)),
            seed=int(cfg.get("seed", 42)),
            subjects=subjects,
        )
        X_tr, yt_tr, y3_tr, sid_tr = parts["train"]
        X_va, yt_va, y3_va, sid_va = parts["val"]

        np.save(out_dir / "train_X.npy", X_tr)
        np.save(out_dir / "train_y_task.npy", yt_tr)
        np.save(out_dir / "train_y_three.npy", y3_tr)
        np.save(out_dir / "train_subjects.npy", sid_tr)

        np.save(out_dir / "val_X.npy", X_va)
        np.save(out_dir / "val_y_task.npy", yt_va)
        np.save(out_dir / "val_y_three.npy", y3_va)
        np.save(out_dir / "val_subjects.npy", sid_va)
        print("saved split → train", X_tr.shape, "val", X_va.shape)

    return out_dir


def main() -> None:
    ap = argparse.ArgumentParser(description="统一预处理：按 yaml 处理整个数据集")
    ap.add_argument("--config", required=True, help="如 configs/bci2a.yaml")
    args = ap.parse_args()

    cfg = load_config(args.config)
    X, y_task, y_three, subjects = process_whole_dataset(cfg)
    save_outputs(X, y_task, y_three, subjects, cfg)
    print("全部完成:", cfg["dataset"])


if __name__ == "__main__":
    main()
```

## 8. 一句话总结

> **一套 steps + 多套 yaml + 多个 loader（registry）** = 统一预处理；  
> **`data_glob` 扫全库 → 逐文件 preprocess → concatenate → 保存/划分** = 自动处理一个数据集里的所有数据；  
> 新数据集 = 新 loader（如需）+ 新 yaml，而不是再复制一整条 pipeline。
