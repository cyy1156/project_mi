# 串联流水线 + 单受试者验收

## 目标

把 Step1–11（及可选 Step12）串起来。对 **A01**（或多人合并）最终应得到：

```text
X.shape == (N, 1, 8, 1000)   # N 个试次，可送进模型
y_task  : (N,)  取值 {0, 1}     # 0=静息, 1=任务
y_three : (N,)  取值 {0, 1, 2}  # 0=空闲, 1=左手, 2=右手
映射一致：静息(0,0) / 左手(1,1) / 右手(1,2)
无 NaN；每个试次、每通道 mean≈0、std≈1（trial-wise z-score 后）
```

文件建议：`preprocess_lab/src/pipeline.py`  
脚本建议：`preprocess_lab/scripts/run_one_subject.py`

## 完整示例代码（可对照实现）

> 导入路径按你本地包结构微调。函数名与当前工程一致：`to_model_tensor`（等价于文档里的 `to_eegnet_tensor`）。

```python
from __future__ import annotations

from pathlib import Path

import numpy as np

from types import ContinuousEEG  # 若与标准库冲突，用: from src.types import ...
from io.load_bci2a_mat import load_bci2a_mat
from steps.select_channels import select_channels
from steps.filter_car import car_reference, notch_and_bandpass
from steps.harmonize_labels import (
    filter_left_right_events,
    extract_rest_cues,
)
from steps.epoch_baseline import (
    slice_epoch,
    baseline_correct,
    classification_window,
    rest_window_with_baseline,
)
from steps.resample_zscore import (
    resample_to_1000,
    trial_zscore,
    to_model_tensor,
)
from steps.split_subjects import split_all_trials


def _append_window(
    xs: list,
    y_task: list,
    y_three: list,
    win: np.ndarray | None,
    lab_task: int,
    lab_three: int,
    fs: float,
) -> None:
    """单个 4s 窗：重采样 → z-score → 收入列表。"""
    if win is None:
        return
    win = resample_to_1000(win, fs_in=fs, fs_out=250.0)
    if win.shape != (1000, 8):
        return
    win = trial_zscore(win)
    xs.append(win)
    y_task.append(lab_task)
    y_three.append(lab_three)


def preprocess_run(
    eeg: ContinuousEEG,
    add_rest: bool = True,
    max_rest: int | None = None,
) -> tuple[np.ndarray, np.ndarray, np.ndarray]:
    """
    处理单个 ContinuousEEG（通常是一个 run）。
    返回: X (N,1,8,1000), y_task (N,), y_three (N,)
    """
    # ----- 连续段：选通道 → CAR → 滤波 -----
    x = select_channels(eeg.x, eeg.ch_names)       # (T, 8)
    x = car_reference(x)
    x = notch_and_bandpass(x, eeg.fs)

    # ----- 左 / 右事件 → 双标签 -----
    kept = filter_left_right_events(eeg.events, eeg.artifacts)
    # kept 每行: [cue, label_task, label_three, trial_index]

    xs: list[np.ndarray] = []
    y_task: list[int] = []
    y_three: list[int] = []

    for cue, lab_task, lab_three, _ in kept:
        ep = slice_epoch(x, int(cue), eeg.fs)
        if ep is None:
            continue
        ep = baseline_correct(ep, eeg.fs)
        win = classification_window(ep, eeg.fs)   # (约1000, 8)
        _append_window(xs, y_task, y_three, win, int(lab_task), int(lab_three), eeg.fs)

    # ----- 静息：下一 Cue 前 4s，标签 (0, 0) -----
    if add_rest and len(kept) > 0:
        starts = extract_rest_cues(kept[:, 0], eeg.fs, x.shape[0])
        # 数量平衡：静息不超过左/右中较少的那一类（可按需要改）
        if max_rest is None:
            n_left = int(np.sum(kept[:, 2] == 1))
            n_right = int(np.sum(kept[:, 2] == 2))
            max_rest = min(n_left, n_right) if (n_left + n_right) else 0
        starts = starts[:max_rest]

        for start in starts:
            win = rest_window_with_baseline(x, int(start), eeg.fs)
            _append_window(xs, y_task, y_three, win, 0, 0, eeg.fs)

    if not xs:
        empty_x = np.zeros((0, 1, 8, 1000), np.float32)
        empty_y = np.zeros((0,), np.int64)
        return empty_x, empty_y, empty_y.copy()

    X = to_model_tensor(xs)  # (N, 1, 8, 1000)
    return (
        X,
        np.asarray(y_task, dtype=np.int64),
        np.asarray(y_three, dtype=np.int64),
    )


def preprocess_subject(
    mat_path: Path,
    add_rest: bool = True,
) -> tuple[np.ndarray, np.ndarray, np.ndarray]:
    """一个 .mat（一名受试者多个 run）→ 合并为 X, y_task, y_three。"""
    Xs, yt, y3 = [], [], []
    for eeg in load_bci2a_mat(mat_path):
        X, y_task, y_three = preprocess_run(eeg, add_rest=add_rest)
        if len(y_task):
            Xs.append(X)
            yt.append(y_task)
            y3.append(y_three)
    if not Xs:
        empty_x = np.zeros((0, 1, 8, 1000), np.float32)
        empty_y = np.zeros((0,), np.int64)
        return empty_x, empty_y, empty_y.copy()
    return (
        np.concatenate(Xs, axis=0),
        np.concatenate(yt, axis=0),
        np.concatenate(y3, axis=0),
    )


def sanity_check_outputs(X, y_task, y_three) -> None:
    assert X.ndim == 4 and X.shape[1:] == (1, 8, 1000)
    assert len(X) == len(y_task) == len(y_three)
    assert set(np.unique(y_task)).issubset({0, 1})
    assert set(np.unique(y_three)).issubset({0, 1, 2})
    assert np.all((y_three == 0) == (y_task == 0))
    assert np.all(y_task[y_three > 0] == 1)
    assert np.isfinite(X).all()
    # 随机抽几个试次看 z-score
    for i in np.random.choice(len(X), size=min(5, len(X)), replace=False):
        trial = X[i, 0]  # (8, 1000)
        assert np.allclose(trial.mean(axis=1), 0, atol=1e-5)
        assert np.allclose(trial.std(axis=1), 1, atol=1e-4)
    print(
        "OK",
        "X", X.shape,
        "y_task", np.bincount(y_task, minlength=2),
        "y_three", np.bincount(y_three, minlength=3),
    )


def main() -> None:
    mat_path = Path(r"D:\360MoveData\Users\ckgxnn\Desktop\MI\DATA\bci2a\A01T.mat")
    out_dir = Path(__file__).resolve().parents[1] / "out"
    out_dir.mkdir(parents=True, exist_ok=True)

    # 1) 单受试者完整预处理
    X, y_task, y_three = preprocess_subject(mat_path, add_rest=True)
    sanity_check_outputs(X, y_task, y_three)

    # 2) 保存全量（未划分）三件套
    np.save(out_dir / "A01_X.npy", X)
    np.save(out_dir / "A01_y_task.npy", y_task)
    np.save(out_dir / "A01_y_three.npy", y_three)
    print("saved full set to", out_dir)

    # 3) 全体试次 8:2 划分（多人时先 concatenate 各人的 X/y/subjects）
    subjects = np.array(["A01"] * len(X))
    parts = split_all_trials(
        X, y_task, y_three, val_ratio=0.2, seed=42, subjects=subjects
    )
    X_tr, yt_tr, y3_tr, sid_tr = parts["train"]
    X_va, yt_va, y3_va, sid_va = parts["val"]
    print("train", X_tr.shape, "val", X_va.shape)

    # 4) 保存训练集 / 验证集（进模型用这套）
    np.save(out_dir / "train_X.npy", X_tr)
    np.save(out_dir / "train_y_task.npy", yt_tr)
    np.save(out_dir / "train_y_three.npy", y3_tr)
    np.save(out_dir / "train_subjects.npy", sid_tr)

    np.save(out_dir / "val_X.npy", X_va)
    np.save(out_dir / "val_y_task.npy", yt_va)
    np.save(out_dir / "val_y_three.npy", y3_va)
    np.save(out_dir / "val_subjects.npy", sid_va)
    print("saved train/val to", out_dir)

    # 以后读取示例：
    # X_tr = np.load(out_dir / "train_X.npy")
    # yt_tr = np.load(out_dir / "train_y_task.npy")
    # y3_tr = np.load(out_dir / "train_y_three.npy")


if __name__ == "__main__":
    main()
```

## 流水线在做什么（对应代码顺序）

```text
A01T.mat
  → load_bci2a_mat          多个 ContinuousEEG（每个正式 run 一条）
  → 对每个 run:
        选 8 通道 → CAR → Notch50 + Bandpass 8–30
        筛左/右（丢脚/舌/伪迹）→ 双标签
        切 Epoch → 基线 → 分类窗 4s
        （可选）造静息窗，标签 (0,0)
        resample → trial z-score
  → 堆成张量 X，对齐 y_task / y_three
  → 保存全量 .npy
  → split_all_trials 得到 train/val
  → 保存 train_*.npy / val_*.npy
```

## 这套流水线是「只对 BCI 2a」还是「所有数据集」？

**目标规格统一，入口按数据集分别写。**

| 部分 | 是否通用 |
|------|----------|
| 最终输出 `(N,1,8,1000)` + 双标签 | ✅ 所有数据集都要落到这套规格 |
| Step3–11（选通道、CAR、滤波、切窗、z-score、张量） | ✅ 只要已是 `ContinuousEEG` 且有目标 8 通道，逻辑通用 |
| Step1 读取 | ❌ 每种格式一个 loader：`load_bci2a_mat` / `load_gdf` / `load_openbci_csv`… |
| Step2 事件码 → 双标签 | ❌ 映射表按数据集改（2a 是 1左2右；别的数据集事件码不同） |
| 静息怎么造 | ⚠️ 2a 用「下一 Cue 前 4s」；别的范式可能已有静息标签或不同间隔 |
| 缺通道的数据集（如 2b 只有 C3/Cz/C4） | ❌ 不能硬套 8 通道标准（不补零） |

**一句话：**  
中间处理和最终张量是**跨数据集统一标准**；当前示例代码是按 **BCI IV 2a `.mat`** 写全的。换 OpenBMI / OpenBCI 时，主要改 **loader + 标签映射**，后面步骤尽量复用。

## 必须遵守的顺序

```text
选通道 → CAR → Notch+Bandpass →（按事件）切 Epoch
→ 基线 → 分类窗 → resample → trial Z-score → 堆张量
（标签：同一试次同时写出 y_task 与 y_three）
```

| 操作 | 位置 |
|------|------|
| CAR | 连续段、滤波前 |
| 基线 | 切 epoch 后 |
| Trial Z-score | 重采样后、进模型前 |

## A01 验收清单

- [ ] `X.shape[1:] == (1, 8, 1000)`
- [ ] `len(X) == len(y_task) == len(y_three)`
- [ ] `set(np.unique(y_task)) ⊆ {0, 1}`
- [ ] `set(np.unique(y_three)) ⊆ {0, 1, 2}`
- [ ] 一致性：`(y_three==0) ↔ (y_task==0)`；`(y_three∈{1,2}) → (y_task==1)`
- [ ] `np.isfinite(X).all()`
- [ ] 随机抽 5 个 trial：各通道 `std ≈ 1`，`mean ≈ 0`
- [ ] 左/右数量大致接近；有静息时说明平衡策略（示例用 `min(n_left,n_right)`）
- [ ] 不要直接拿旧四分类 `epochs_npy` 当本流水线输出
