## 0. 你要改成什么样

### 现在

```text
A01T.mat → 多 run → 合并 → train/val npy（只有 A01）
```

### 目标（批处理）

```text
data_glob: DATA/bci2a/A0*T.mat
    → [A01T, A02T, …, A09T]
    → 每人：多 run → preprocess_run
    → 全体 concatenate（带 subjects）
    → 保存全量 + 全体试次 8:2 → train/val
```

训练侧仍读：

```text
train_X.npy / train_y_task.npy / …
val_X.npy   / val_y_task.npy   / …
```

只是 N 从「一人约 200」变成「九人合计约两千级」（视伪迹/静息而定）。

## 3. `config_load.py`（完整示例）

**路径：** `code/preprocess_lab/src/config_load.py`

```python
# 文件: code/preprocess_lab/src/config_load.py
from __future__ import annotations

from pathlib import Path

import yaml


def load_config(path: str | Path) -> dict:
    path = Path(path)
    with path.open("r", encoding="utf-8") as f:
        cfg = yaml.safe_load(f)
    if not isinstance(cfg, dict):
        raise ValueError(f"配置应为映射表: {path}")
    # 相对 out_dir：相对「配置文件所在目录的上一级」= preprocess_lab/
    out = Path(cfg["out_dir"])
    if not out.is_absolute():
        lab_root = path.resolve().parent.parent  # configs/ → preprocess_lab/
        cfg["out_dir"] = str((lab_root / out).resolve())
    return cfg
```

需已安装：`pyyaml`（你 `code/requirements.txt` 里已有）。

```python
# 文件: code/preprocess_lab/src/io/registry.py
from __future__ import annotations

from pathlib import Path
from typing import Callable

from src.io.load_bci2a_mat import load_bci2a_mat
from src.types import ContinuousEEG

LoaderFn = Callable[[Path], list[ContinuousEEG] | ContinuousEEG]

LOADERS: dict[str, LoaderFn] = {
    "load_bci2a_mat": load_bci2a_mat,
    # 以后:
    # "load_openbci_csv": load_openbci_csv,
}


def get_loader(name: str) -> LoaderFn:
    if name not in LOADERS:
        raise KeyError(f"未知 loader: {name}，可选: {list(LOADERS)}")
    return LOADERS[name]


def as_run_list(obj: list[ContinuousEEG] | ContinuousEEG) -> list[ContinuousEEG]:
    if isinstance(obj, list):
        return obj
    return [obj]
```

## 5. 批处理主程序（完整示例）

**路径：** `code/preprocess_lab/src/pipeline_batch.py`  
（新建；保留旧 `pipeline.py` 专跑 A01 调试。）

下面直接复用你现有的 `preprocess_run`（从 `pipeline` 导入）。若你更希望「配置驱动滤波参数」，可再包一层；**第一期批处理的最小目标是：多文件循环 + 合并 + 保存**。

```python
# 文件: code/preprocess_lab/src/pipeline_batch.py
from __future__ import annotations

import argparse
import glob
from pathlib import Path

import numpy as np

from src.config_load import load_config
from src.io.registry import as_run_list, get_loader
from src.pipeline import preprocess_run, sanity_check_outputs
from src.steps.split_subjects import split_all_trials


def collect_files(cfg: dict) -> list[Path]:
    """根据 yaml 收集本数据集全部输入文件。"""
    if cfg.get("data_files"):
        files = [Path(p) for p in cfg["data_files"]]
    else:
        pattern = cfg["data_glob"]
        files = sorted(Path(p) for p in glob.glob(pattern))
    if not files:
        raise FileNotFoundError(
            f"没有匹配到数据文件，请检查 data_glob / data_files。dataset={cfg.get('dataset')}"
        )
    print(f"[{cfg['dataset']}] 共 {len(files)} 个文件:")
    for f in files:
        print(" ", f)
    return files


def subject_id_from_path(path: Path, cfg: dict) -> str:
    mode = cfg.get("subject_from", "stem3")
    if mode == "stem3":
        return path.stem[:3]  # A01T → A01
    if mode == "stem":
        return path.stem
    raise ValueError(f"未知 subject_from: {mode}")


def process_one_file(
    path: Path,
    cfg: dict,
) -> tuple[np.ndarray, np.ndarray, np.ndarray, np.ndarray]:
    """
    处理一个 .mat：内部多 run → 合并。
    返回 X, y_task, y_three, subjects（与试次等长）。
    """
    loader = get_loader(cfg["loader"])
    sid = subject_id_from_path(path, cfg)
    runs = as_run_list(loader(path))
    print(f"  → {path.name}: subject={sid}, runs={len(runs)}")

    Xs, yts, y3s = [], [], []
    add_rest = bool(cfg.get("make_rest", True))

    for eeg in runs:
        # 若 ContinuousEEG 已带 subject，仍以文件名为准更稳
        X, yt, y3 = preprocess_run(eeg, add_rest=add_rest)
        if len(yt) == 0:
            continue
        Xs.append(X)
        yts.append(yt)
        y3s.append(y3)

    if not Xs:
        empty = np.zeros((0, 1, 8, 1000), np.float32)
        z = np.zeros((0,), np.int64)
        return empty, z, z.copy(), np.array([], dtype=object)

    X = np.concatenate(Xs, axis=0)
    y_task = np.concatenate(yts, axis=0)
    y_three = np.concatenate(y3s, axis=0)
    subjects = np.array([sid] * len(y_task), dtype=object)
    return X, y_task, y_three, subjects


def process_whole_dataset(
    cfg: dict,
) -> tuple[np.ndarray, np.ndarray, np.ndarray, np.ndarray]:
    """扫配置中的全部文件并合并。"""
    Xs, yts, y3s, subs = [], [], [], []

    for fpath in collect_files(cfg):
        X, yt, y3, sid = process_one_file(fpath, cfg)
        if len(yt) == 0:
            print(f"  ⚠ 跳过（无有效试次）: {fpath.name}")
            continue
        print(
            f"  完成 {fpath.name}: X={X.shape}, "
            f"task={np.bincount(yt, minlength=2)}, "
            f"three={np.bincount(y3, minlength=3)}"
        )
        Xs.append(X)
        yts.append(yt)
        y3s.append(y3)
        subs.append(sid)

    if not Xs:
        raise RuntimeError("全部文件处理后没有任何有效试次")

    X = np.concatenate(Xs, axis=0)
    y_task = np.concatenate(yts, axis=0)
    y_three = np.concatenate(y3s, axis=0)
    subjects = np.concatenate(subs, axis=0)

    print("=" * 50)
    print(
        "合并完成:",
        "X", X.shape,
        "y_task", np.bincount(y_task, minlength=2),
        "y_three", np.bincount(y_three, minlength=3),
        "subjects", sorted(set(subjects.tolist())),
    )
    sanity_check_outputs(X, y_task, y_three)
    return X, y_task, y_three, subjects


def save_outputs(
    X: np.ndarray,
    y_task: np.ndarray,
    y_three: np.ndarray,
    subjects: np.ndarray,
    cfg: dict,
) -> Path:
    out_dir = Path(cfg["out_dir"])
    out_dir.mkdir(parents=True, exist_ok=True)
    tag = cfg["dataset"]

    if cfg.get("save_full", True):
        np.save(out_dir / f"{tag}_X.npy", X)
        np.save(out_dir / f"{tag}_y_task.npy", y_task)
        np.save(out_dir / f"{tag}_y_three.npy", y_three)
        np.save(out_dir / f"{tag}_subjects.npy", subjects)
        print("saved full →", out_dir)

    if cfg.get("save_split", True):
        parts = split_all_trials(
            X,
            y_task,
            y_three,
            val_ratio=float(cfg.get("val_ratio", 0.2)),
            seed=int(cfg.get("seed", 42)),
            subjects=subjects,
        )
        X_tr, yt_tr, y3_tr, sid_tr = parts["train"]
        X_va, yt_va, y3_va, sid_va = parts["val"]

        # 文件名与当前 train_lab 约定一致，便于直接训练
        np.save(out_dir / "train_X.npy", X_tr)
        np.save(out_dir / "train_y_task.npy", yt_tr)
        np.save(out_dir / "train_y_three.npy", y3_tr)
        np.save(out_dir / "train_subjects.npy", sid_tr)

        np.save(out_dir / "val_X.npy", X_va)
        np.save(out_dir / "val_y_task.npy", yt_va)
        np.save(out_dir / "val_y_three.npy", y3_va)
        np.save(out_dir / "val_subjects.npy", sid_va)
        print("saved split → train", X_tr.shape, "val", X_va.shape)

    return out_dir


def main() -> None:
    ap = argparse.ArgumentParser(description="整库批处理预处理")
    ap.add_argument(
        "--config",
        default="configs/bci2a.yaml",
        help="相对 preprocess_lab 的配置路径，或绝对路径",
    )
    args = ap.parse_args()

    cfg_path = Path(args.config)
    if not cfg_path.is_absolute():
        # 默认相对 preprocess_lab 根目录
        lab_root = Path(__file__).resolve().parents[1]
        cfg_path = lab_root / cfg_path

    cfg = load_config(cfg_path)
    X, y_task, y_three, subjects = process_whole_dataset(cfg)
    save_outputs(X, y_task, y_three, subjects, cfg)
    print("全部完成:", cfg["dataset"], "→", cfg["out_dir"])


if __name__ == "__main__":
    main()
```

## 7. 和训练代码怎么接

你现在的训练读的是：

```text
code/preprocess_lab/out/train_X.npy
```

批处理若写到 `out/bci2a/`，需要二选一（**改训练路径或改配置 out_dir**）：

| 方案 | 做法 |
|------|------|
| A | yaml 里设 `out_dir: "out"`（仍写到原来的 `preprocess_lab/out/`） |
| B | 训练里 `DATA_DIR` 改成 `.../preprocess_lab/out/bci2a` |

推荐调试期用 **方案 A**，少改训练；正式实验用 `out/bci2a` 分数据集归档（方案 B）。

## 8. 相对旧 `pipeline.py` 你改了什么（对照）

| 点 | 旧 `main()` | 批处理示例 |
|----|-------------|------------|
| 输入 | 写死 `A01T.mat` | `data_glob` / `data_files` |
| 循环 | 无（单文件） | `for fpath in collect_files` |
| subject | 手写 `["A01"]*N` | 每文件 `stem3`，合并进 `subjects` |
| 核心算法 | `preprocess_run` | **继续调用**，不重写滤波切窗 |
| 划分 | `split_all_trials` 8:2 | 相同，但是在**全库合并后**再划 |

**不必**为了批处理重写 Step3–11；批处理 = **外层多文件 for + 合并保存**。

## 10. 常见坑

| 现象 | 原因 | 处理 |
|------|------|------|
| `没有匹配到数据文件` | `data_glob` 路径/通配错 | 先 `print(glob.glob(...))` |
| 某人 0 试次 | 伪迹多或 run 跳过 | 日志会「跳过」；检查该 mat |
| 训练找不到 npy | `out_dir` 与 `DATA_DIR` 不一致 | 见 §7 |
| 内存紧张 | 九人一次拼很大 | 可先 2～3 人；或以后改「按人存再合并」 |
| 与旧 `pipeline` 冲突 | 两套入口 | 调试用旧文件，批处理用 `pipeline_batch` |
